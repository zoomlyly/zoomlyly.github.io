js效果--按钮倒计时

html部分:

css部分:

.button-green { background-color: #3fb043; } .button-nogreen { background-color: #D7DCD7; }

javascript部分:

var btnObject= document.getElementById('btn');
btnObject.onclick=function(){
 btnObject.className='button-nogreen';
    btnObject.disabled='disabled';
    var wait = 60;
    var waiter = null;
    waiter = setInterval(function() {
        wait--;
        btnObject.innerHTML = wait + '后重新发送';
        if (wait == 0) {
            clearInterval(waiter);
            btnObject.innerHTML = '免费获取验证码';
            btnObject.className='button-green';
            btnObject.disabled='';
        }
    }, 1000);
}
